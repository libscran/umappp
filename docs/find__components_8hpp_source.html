28/* Finds the connected components of the graph. This assumes that
29 * the edges are already symmetrized by combine_neighbor_sets,
30 * allowing us to use a single-pass through the edge list.
31 */
32inline ComponentIndices find_components(const NeighborList& edges) {
33 int counter = 0;
34 std::vector<int> mapping(edges.size(), -1);
35
36 for (size_t current = 0; current < edges.size(); ++current) {
37 if (mapping[current] != -1) {
38 continue;
39 }
40
41 std::vector<int> remaining(1, current);
42 mapping[current] = counter;
43 do {
44 int curfriend = remaining.back();
45 remaining.pop_back();
46
47 for (const auto& ff : edges[curfriend]) {
48 if (mapping[ff.first] == -1) {
49 remaining.push_back(ff.first);
50 mapping[ff.first] = counter;
51 }
52 }
53 } while (remaining.size());
54
55 ++counter;
56 }
57
58 return ComponentIndices(counter, std::move(mapping));
59}
